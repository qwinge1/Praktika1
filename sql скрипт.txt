-- ============================================================
-- 3. Функция преобразования русской даты
-- ============================================================
CREATE FUNCTION [dbo].[parse_russian_date] (@date_str NVARCHAR(100))
RETURNS DATE
AS
BEGIN
    DECLARE @day INT, @month_name NVARCHAR(20), @year INT, @month_num INT;
    
    -- Удаляем лишние пробелы и разбиваем строку
    SET @date_str = LTRIM(RTRIM(@date_str));
    
    -- Предполагаем формат: "18 октября 1986 года" или "9 октября 1977 года"
    DECLARE @space1 INT = CHARINDEX(' ', @date_str);
    DECLARE @space2 INT = CHARINDEX(' ', @date_str, @space1 + 1);
    
    SET @day = CAST(LEFT(@date_str, @space1 - 1) AS INT);
    SET @month_name = SUBSTRING(@date_str, @space1 + 1, @space2 - @space1 - 1);
    SET @year = CAST(REPLACE(SUBSTRING(@date_str, @space2 + 1, LEN(@date_str)), 'года', '') AS INT);
    
    SET @month_num = CASE LOWER(@month_name)
        WHEN N'января' THEN 1
        WHEN N'февраля' THEN 2
        WHEN N'марта' THEN 3
        WHEN N'апреля' THEN 4
        WHEN N'мая' THEN 5
        WHEN N'июня' THEN 6
        WHEN N'июля' THEN 7
        WHEN N'августа' THEN 8
        WHEN N'сентября' THEN 9
        WHEN N'октября' THEN 10
        WHEN N'ноября' THEN 11
        WHEN N'декабря' THEN 12
        ELSE NULL
    END;
    
    RETURN DATEFROMPARTS(@year, @month_num, @day);
END;

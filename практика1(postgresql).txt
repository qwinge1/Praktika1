CREATE TABLE Роль (
    id SERIAL PRIMARY KEY,
    название VARCHAR(50) NOT NULL UNIQUE
);
CREATE TABLE Подразделение (
    id SERIAL PRIMARY KEY,
    название VARCHAR(255) NOT NULL UNIQUE,
    описание TEXT
);
CREATE TABLE Сотрудник (
    id SERIAL PRIMARY KEY,
    фамилия VARCHAR(100) NOT NULL,
    имя VARCHAR(100) NOT NULL,
    отчество VARCHAR(100),
    подразделение_id INT REFERENCES Подразделение(id) ON DELETE SET NULL,
    табельный_номер VARCHAR(20) UNIQUE,
    активен BOOLEAN DEFAULT TRUE
);
CREATE TABLE Пользователь (
    id SERIAL PRIMARY KEY,
    email VARCHAR(255) NOT NULL UNIQUE,
    хеш_пароля VARCHAR(255) NOT NULL,
    роль_id INT NOT NULL REFERENCES Роль(id),
    дата_регистрации TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE Статус (
    id SERIAL PRIMARY KEY,
    название VARCHAR(50) NOT NULL UNIQUE
);
CREATE TABLE Заявка (
    id SERIAL PRIMARY KEY,
    пользователь_id INT NOT NULL REFERENCES Пользователь(id),
    тип VARCHAR(20) NOT NULL CHECK (тип IN ('личная', 'групповая')),
    дата_начала DATE NOT NULL,
    дата_окончания DATE NOT NULL,
    цель_посещения TEXT NOT NULL,
    подразделение_id INT NOT NULL REFERENCES Подразделение(id),
    сотрудник_id INT NOT NULL REFERENCES Сотрудник(id),
    статус_id INT NOT NULL REFERENCES Статус(id),
    причина_отказа TEXT,
    дата_создания TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    дата_обновления TIMESTAMP
);
CREATE TABLE ПосетительЛичный (
    id SERIAL PRIMARY KEY,
    заявка_id INT NOT NULL UNIQUE REFERENCES Заявка(id) ON DELETE CASCADE,
    фамилия VARCHAR(100) NOT NULL,
    имя VARCHAR(100) NOT NULL,
    отчество VARCHAR(100),
    телефон VARCHAR(20),
    email VARCHAR(255) NOT NULL,
    организация VARCHAR(255),
    примечание TEXT NOT NULL,
    дата_рождения DATE NOT NULL,
    серия_паспорта CHAR(4) NOT NULL,
    номер_паспорта CHAR(6) NOT NULL,
    путь_к_фото VARCHAR(500)
);
CREATE TABLE ПосетительГрупповой (
    id SERIAL PRIMARY KEY,
    заявка_id INT NOT NULL REFERENCES Заявка(id) ON DELETE CASCADE,
    номер_по_порядку INT NOT NULL,
    фамилия VARCHAR(100) NOT NULL,
    имя VARCHAR(100) NOT NULL,
    отчество VARCHAR(100),
    телефон VARCHAR(20),
    email VARCHAR(255) NOT NULL,
    организация VARCHAR(255),
    примечание TEXT NOT NULL,
    дата_рождения DATE NOT NULL,
    серия_паспорта CHAR(4) NOT NULL,
    номер_паспорта CHAR(6) NOT NULL,
    путь_к_фото VARCHAR(500)
);
CREATE TABLE Документ (
    id SERIAL PRIMARY KEY,
    посетитель_личный_id INT REFERENCES ПосетительЛичный(id) ON DELETE CASCADE,
    посетитель_групповой_id INT REFERENCES ПосетительГрупповой(id) ON DELETE CASCADE,
    тип_документа VARCHAR(20) NOT NULL CHECK (тип_документа IN ('скан_паспорта', 'фото')),
    путь_к_файлу VARCHAR(500) NOT NULL,
    дата_загрузки TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CHECK (посетитель_личный_id IS NOT NULL OR посетитель_групповой_id IS NOT NULL)
);
CREATE TABLE ЖурналПроходов (
    id SERIAL PRIMARY KEY,
    заявка_id INT NOT NULL REFERENCES Заявка(id),
    посетитель_личный_id INT REFERENCES ПосетительЛичный(id) ON DELETE SET NULL,
    посетитель_групповой_id INT REFERENCES ПосетительГрупповой(id) ON DELETE SET NULL,
    время_входа TIMESTAMP NOT NULL,
    время_выхода TIMESTAMP,
    сотрудник_охраны_id INT NOT NULL REFERENCES Сотрудник(id),
    CHECK (посетитель_личный_id IS NOT NULL OR посетитель_групповой_id IS NOT NULL)
);

CREATE OR REPLACE FUNCTION parse_russian_date(date_str TEXT) RETURNS DATE AS $$
DECLARE
    parts TEXT[];
    day INT;
    month_name TEXT;
    year INT;
    month_num INT;
BEGIN
    parts := regexp_split_to_array(trim(date_str), '\s+');
    day := parts[1]::INT;
    month_name := lower(parts[2]);
    year := regexp_replace(parts[3], '[^0-9]', '', 'g')::INT;
    month_num := CASE month_name
        WHEN 'января' THEN 1 WHEN 'февраля' THEN 2 WHEN 'марта' THEN 3
        WHEN 'апреля' THEN 4 WHEN 'мая' THEN 5 WHEN 'июня' THEN 6
        WHEN 'июля' THEN 7 WHEN 'августа' THEN 8 WHEN 'сентября' THEN 9
        WHEN 'октября' THEN 10 WHEN 'ноября' THEN 11 WHEN 'декабря' THEN 12
        ELSE NULL
    END;
    IF month_num IS NULL THEN RAISE EXCEPTION 'Неизвестный месяц: %', month_name; END IF;
    RETURN make_date(year, month_num, day);
END;
$$ LANGUAGE plpgsql;

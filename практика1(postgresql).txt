CREATE TABLE Роль (
    id SERIAL PRIMARY KEY,
    название VARCHAR(50) NOT NULL UNIQUE

CREATE TABLE Подразделение (
    id SERIAL PRIMARY KEY,
    название VARCHAR(255) NOT NULL UNIQUE,

CREATE TABLE Сотрудник (
    id SERIAL PRIMARY KEY,
    фамилия VARCHAR(100) NOT NULL,
    имя VARCHAR(100) NOT NULL,
    отчество VARCHAR(100),
    подразделение_id INT REFERENCES Подразделение(id) ON DELETE SET NULL,

CREATE TABLE Пользователь (
    id SERIAL PRIMARY KEY,
    email VARCHAR(255) NOT NULL UNIQUE,
    хеш_пароля VARCHAR(255) NOT NULL,

CREATE TABLE Статус (

CREATE TABLE ЧёрныйСписок (
    id SERIAL PRIMARY KEY,
    тип_посетителя VARCHAR(20) NOT NULL CHECK (тип_посетителя IN ('личный', 'групповой')),
    посетитель_личный_id INT REFERENCES ПосетительЛичный(id) ON DELETE CASCADE,

        (тип_посетителя = 'личный' AND посетитель_личный_id IS NOT NULL AND посетитель_групповой_id IS NULL) OR

-- 3. Функция преобразования русской даты
-- ============================================================
CREATE OR REPLACE FUNCTION parse_russian_date(date_str TEXT) RETURNS DATE AS $$
DECLARE
    parts TEXT[];
    day INT;
    month_name TEXT;
    year INT;
    month_num INT;
BEGIN
    parts := regexp_split_to_array(trim(date_str), '\s+');
    day := parts[1]::INT;
    month_name := lower(parts[2]);
    year := regexp_replace(parts[3], '[^0-9]', '', 'g')::INT;
    
    month_num := CASE month_name
        WHEN 'января' THEN 1   WHEN 'февраля' THEN 2  WHEN 'марта' THEN 3
        WHEN 'апреля' THEN 4   WHEN 'мая' THEN 5      WHEN 'июня' THEN 6
        WHEN 'июля' THEN 7     WHEN 'августа' THEN 8  WHEN 'сентября' THEN 9
        WHEN 'октября' THEN 10 WHEN 'ноября' THEN 11  WHEN 'декабря' THEN 12
        ELSE NULL
    END;
    
    IF month_num IS NULL THEN
        RAISE EXCEPTION 'Неизвестный месяц: %', month_name;
    END IF;
    
    RETURN make_date(year, month_num, day);
END;
$$ LANGUAGE plpgsql;

-- ============================================================
-- 4. Триггер для генерации логина пользователя
-- ============================================================
CREATE OR REPLACE FUNCTION generate_user_login_after()
RETURNS TRIGGER AS $$
DECLARE
    at_pos INT;
    email_prefix TEXT;
    new_login TEXT;
BEGIN
    at_pos := POSITION('@' IN NEW.email);
    IF at_pos > 1 THEN
        email_prefix := LEFT(NEW.email, at_pos - 1);
    ELSE
        email_prefix := NEW.email;
    END IF;

    new_login := email_prefix || NEW.id;

    UPDATE Пользователь SET login = new_login WHERE id = NEW.id;

    RAISE NOTICE 'Пользователю % сгенерирован логин: %', NEW.email, new_login;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER user_after_insert
    AFTER INSERT ON Пользователь
    FOR EACH ROW
    EXECUTE FUNCTION generate_user_login_after();

-- 5. Представление ViewListRequests
-- ============================================================
CREATE OR REPLACE VIEW ViewListRequests AS
SELECT 
    z.id AS заявка_id,
    z.тип AS тип_заявки,
    z.дата_начала,
    z.дата_окончания,
    z.цель_посещения,
    z.время_посещения,
    z.причина_отказа,
    z.дата_создания,
    z.дата_обновления,
    p.название AS подразделение,
    s.название AS статус,
    u.email AS email_пользователя,
    pl.фамилия AS фамилия_личн,
    pl.имя AS имя_личн,
    pl.отчество AS отчество_личн,
    pl.телефон AS телефон_личн,
    pl.email AS email_личн,
    pl.организация AS организация_личн,
    pl.примечание AS примечание_личн,
    pl.дата_рождения AS дата_рождения_личн,
    pl.серия_паспорта AS серия_паспорта_личн,
    pl.номер_паспорта AS номер_паспорта_личн,
    pl.путь_к_фото AS фото_личн,
    string_agg(DISTINCT pg.фамилия || ' ' || pg.имя || ' ' || COALESCE(pg.отчество,''), ', ') AS участники_группы,
    COUNT(DISTINCT pg.id) AS кол_во_участников
FROM Заявка z
JOIN Подразделение p ON z.подразделение_id = p.id
JOIN Статус s ON z.статус_id = s.id
JOIN Пользователь u ON z.пользователь_id = u.id
LEFT JOIN ПосетительЛичный pl ON z.id = pl.заявка_id AND z.тип = 'личная'
LEFT JOIN ПосетительГрупповой pg ON z.id = pg.заявка_id AND z.тип = 'групповая'
GROUP BY 
    z.id, p.название, s.название, u.email,
    pl.фамилия, pl.имя, pl.отчество, pl.телефон, pl.email, pl.организация,
    pl.примечание, pl.дата_рождения, pl.серия_паспорта, pl.номер_паспорта, pl.путь_к_фото;

-- ============================================================
-- 6. Функция фильтрации заявок FilteringRequests
-- ============================================================
CREATE OR REPLACE FUNCTION FilteringRequests(
    p_тип VARCHAR DEFAULT NULL,
    p_подразделение_id INT DEFAULT NULL,
    p_статус_id INT DEFAULT NULL
)
RETURNS TABLE(
    заявка_id INT,
    тип_заявки VARCHAR,
    дата_начала DATE,
    дата_окончания DATE,
    цель_посещения TEXT,
    время_посещения TIMESTAMP,
    подразделение VARCHAR,
    статус VARCHAR,
    email_пользователя VARCHAR,
    фамилия_личн VARCHAR,
    имя_личн VARCHAR,
    отчество_личн VARCHAR,
    участники_группы TEXT,
    кол_во_участников BIGINT
) AS $$
BEGIN
    RETURN QUERY
    SELECT 
        v.заявка_id,
        v.тип_заявки,
        v.дата_начала,
        v.дата_окончания,
        v.цель_посещения,
        v.время_посещения,
        v.подразделение,
        v.статус,
        v.email_пользователя,
        v.фамилия_личн,
        v.имя_личн,
        v.отчество_личн,
        v.участники_группы,
        v.кол_во_участников
    FROM ViewListRequests v
    WHERE 
        (p_тип IS NULL OR v.тип_заявки = p_тип) AND
        (p_подразделение_id IS NULL OR EXISTS (
            SELECT 1 FROM Заявка z WHERE z.id = v.заявка_id AND z.подразделение_id = p_подразделение_id
        )) AND
        (p_статус_id IS NULL OR EXISTS (
            SELECT 1 FROM Заявка z WHERE z.id = v.заявка_id AND z.статус_id = p_статус_id
        ))
    ORDER BY v.дата_начала DESC;
END;
$$ LANGUAGE plpgsql;

-- 7. Хранимые процедуры для авторизации/регистрации
-- ============================================================
CREATE OR REPLACE FUNCTION login_user(p_email VARCHAR, p_hash VARCHAR)
RETURNS TABLE(id INT, email VARCHAR, роль_id INT) AS $$
BEGIN
    RETURN QUERY SELECT u.id, u.email, u.роль_id FROM Пользователь u
                 WHERE u.email = p_email AND u.хеш_пароля = p_hash;
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION register_user(p_email VARCHAR, p_hash VARCHAR)
RETURNS VOID AS $$
BEGIN
    INSERT INTO Пользователь (email, хеш_пароля, роль_id)
    VALUES (p_email, p_hash, (SELECT id FROM Роль WHERE название = 'Гость'));
END;
$$ LANGUAGE plpgsql;

            v_security_id);
    
    SELECT з.id, пг.id INTO v_app_групповая, v_visitor_групповой
    FROM Заявка з
    JOIN ПосетительГрупповой пг ON пг.заявка_id = з.id

            v_security_id);

    JOIN ПосетительГрупповой пг ON пг.заявка_id = з.id

            v_security_id);
